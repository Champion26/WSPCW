1 /**
  2 * @description Function designed to send a variable (typically a string url) to a file.
  3 * -Based on examples provided by Kit Lester.
  4 * @param {String} URL
  5 * @param {Unknown} callback
  6 */
  7 function postJSON(variable, URL, callback) {
  8   var ajaxObj = new XMLHttpRequest();
  9   ajaxObj.open("POST", URL, true);
 10   console.log("posting");
 11   ajaxObj.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
 12 
 13   ajaxObj.addEventListener("load",
 14     function() {
 15         console.log("received");
 16         callback(ajaxObj.responseText);
 17     }
 18   );
 19   ajaxObj.send(variable);
 20 }

 22 /**
 23 * @description Function designed to send a variable (typically a string url) to a file and then parse over the JSON response.
 24 * Typically used for sending and immediately retrieving information from PHP responders.
 25 * -Based on examples provided by Kit Lester.
 26 * @param {String} URL
 27 * @param {Unknown} callback
 28 */
 29 function returnJSON(variable, URL, callback) {
 30   var ajaxObj = new XMLHttpRequest();
 31   ajaxObj.open("POST", URL, true);
 32   console.log("posting");
 33   ajaxObj.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
 34 
 35   ajaxObj.addEventListener("load",
 36     function() {
 37         console.log("received");
 38         callback(JSON.parse(ajaxObj.responseText));
 39     }
 40   );
 41   ajaxObj.send(variable);
 42 }
