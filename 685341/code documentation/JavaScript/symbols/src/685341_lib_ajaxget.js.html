1 /**
  2 * @description This function is used to load elements onto the page to give the illusion
  3 that the page has changed.
  4 -Based on examples provided by Kit Lester.
  5 * @param {String} URL
  6 * @param {Unknown} Callback
  7 */
  8 function ajaxGet(URL, callback) {
  9   ifSend();
 10   searchArray = [];
 11   var ajaxObj = new XMLHttpRequest();
 12   ajaxObj.open("GET", URL, true); // The TRUE implies asynchronous
 13   ajaxObj.onreadystatechange = function() {
 14     if (ajaxObj.status === 200){
 15       if (ajaxObj.readyState === 4){
 16         callback(ajaxObj.responseText);
 17       //  setAllContent();
 18         findEditableElements();
 19         setColour(pageColour, textColour, navColour, headingColour, imageSize, navHeight, navWidth);
 20 
 21   }
 22 }
 23   };
 24   ajaxObj.send(null);
 25   findEditableElements();
 26 
 27 }
