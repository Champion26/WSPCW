1 /**
  2 * @description Function designed to retrieve a JSON object from a file and parse over it.
  3 * -Based on examples provided by Kit Lester.
  4 * @param {String} URL
  5 * @param {Unknown} callback
  6 */
  7 function getJSON(URL, callback) {
  8   var ajaxObj = new XMLHttpRequest();
  9 
 10   ajaxObj.open("GET", URL, true); // The TRUE implies asynchronous
 11   ajaxObj.onreadystatechange = function() {
 12     if (ajaxObj.status === 200)
 13       if (ajaxObj.readyState === 4)
 14         callback(JSON.parse(ajaxObj.responseText));
 15       };
 16       ajaxObj.send();
 17 
 18 };
